def detect_outliers(data):
    outliers = []
    threshold = 3  # Standard threshold for z-score
    mean = np.mean(data)
    std = np.std(data)
    
    for i in data:
        z_score = (i - mean) / std
        if np.abs(z_score) > threshold:
            outliers.append(i)
    return outliers

Why there was a spike in 2005 ?¶

Over the years leading up to 2005, capital gains taxes were significantly reduced, affecting income distribution as these tax reductions disproportionately benefited higher-income groups, who derive a larger share of their income from capital gains compared to lower-income groups.

# Calculate the sum of male and female suicides for each country
suicide_sums = dataset.groupby('Country')[['Number of suicides - male', 'Number of suicides - female']].sum()

# Calculate the ratio of male to female suicides for each country
suicide_sums['Ratio M/F'] = suicide_sums['Number of suicides - male'] / suicide_sums['Number of suicides - female']

# Calculate the average Gini coefficient for each country
avg_gini_by_country = dataset.groupby('Country')['Gini coefficient'].mean()

# Join the two datasets
combined_data = suicide_sums.join(avg_gini_by_country)

# Analyze the relationship between Gini coefficient and the ratio of male to female suicides
slope, intercept, r_value, p_value, std_err = linregress(combined_data['Gini coefficient'], combined_data['Ratio M/F'])

# Testing the hypothesis
print(f"Slope: {slope}")
print(f"Intercept: {intercept}")
print(f"R-squared: {r_value**2}")
print(f"P-value: {p_value}")
